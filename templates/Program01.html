Third problem

params.maxiter = 250;
ns = [n, n^2, n^3];
for i = 1:length(ns)
    A = O' * diag([linspace(1, ns(i)-1, n-1), ns(i)]) * O;
    fns = ProbEucQuadratic(A, b, fns);
    [~, info] = RSD(fns, params);
    excact_x = linsolve(A, b);

    precision = [];
    for k = 1:length(info.xs)
        precision = [precision; norm(info.xs(:, k) - excact_x)];
    end
    precisions(:, i) = precision;

end

xdomain = 0:params.maxiter;
myPlot(@semilogy, xdomain, precisions(:, 1), 'LineStyle', '-')
hold on

myPlot(@semilogy, xdomain, precisions(:, 2), 'LineStyle', '--')
myPlot(@semilogy, xdomain, precisions(:, 3), 'LineStyle', '-.')
hold off

xlabel('k', 'FontSize', 16);
ylabel('$\|x_k - x^{*}\|$', 'Interpreter', 'latex', 'FontSize', 16);
legend({['cond=', num2str(ns(1))], ['cond=', num2str(ns(2))], ...
    ['cond=', num2str(ns(3))]});
